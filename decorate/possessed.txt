Actor PossessedMonster : PlayerPawn
{
	-PICKUP
	+NOSKIN
	+NOTARGET
	-FRIENDLY
	
	DamageFactor "Monster", 0
	DamageFactor "Slime", 0
	
	States
	{
	Death:
		NULL A 1
		Loop
	}
}

Actor PossessedWeapon : Weapon
{
	+UNDROPPABLE

	States
	{
	Ready:
		TNT1 A 1 A_WeaponReady
		Loop
	Select:
		TNT1 A 0 A_Raise
		Loop
	Deselect:
		TNT1 A 0 A_Lower
		Loop
	Fire:
		TNT1 A 0
		Goto Ready
	Spawn:
		TNT1 A -1
		Stop
	}
}
