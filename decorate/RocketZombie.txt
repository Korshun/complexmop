ACTOR PossessedRocketZombie : PossessedMonster
{ 
	Player.MorphWeapon "RocketZombieWeapon"

	Health 80
	Radius 20
	Height 56
	Speed 0.35
	Mass 100
	PainChance 100
	PainChance "ZombieRocket", 0
	DropItem "PortableMedkit" 120
	DropItem "ArmorPlate" 120
	DropItem "AmmoPack" 80
	DropItem "Missile" 256 5
	DropItem "HandGrenadeAmmo" 200 1
	DropItem "MineAmmo" 100 1
	DamageFactor "ZombieRocket", 0.2
	+FLOORCLIP
	+MISSILEMORE
	SeeSound "grunt/sight"
	PainSound "grunt/pain"
	DeathSound "grunt/death"
	ActiveSound "grunt/active"
	Obituary "%o ate a Zombie's rocket."
	States
	{
	Spawn:
		MPO2 A 10
		MPOI A 10
		Loop
	See:
		MPO2 AABB 3
		TNT1 A 0 A_SpawnItem("FootStep",0,0,0,0)
		MPO2 CCDD 3
		TNT1 A 0 A_SpawnItem("FootStep",0,0,0,0)
		Loop
	Missile:
		MPO2 F 6 Bright
		MPO2 E 20 A_PlaySound("rocket/drumspin")
		MPO2 E 10
		Goto See
    }
}

Actor RocketZombieWeapon : PossessedWeapon
{
	States
	{
	Fire:
		TNT1 A 6 Bright A_FireCustomMissile("ZombieRocket")
		TNT1 A 20 A_PlaySound("rocket/drumspin")
		Goto Ready
	}
}
