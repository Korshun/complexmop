ACTOR PossessedShotgunZombie : PossessedMonster
{ 
	Player.MorphWeapon "ShotgunZombieWeapon"

	Health 30
	Radius 20
	Height 56
	Speed 0.35
	Mass 100
	PainChance 160
	DropItem "HealthFlask" 35
	DropItem "ArmorCharge" 35
	DropItem "AmmoPack" 20
	DropItem "HandGrenadeAmmo" 50 1
	DropItem "MineAmmo" 25 1
	DropItem "Combat Shotgun" 256 1
	MONSTER 
	+FLOORCLIP
	+MISSILEMORE
	SeeSound "grunt/sight"
	AttackSound "weapons/12gaugefire"
	PainSound "grunt/pain"
	DeathSound "grunt/death"
	ActiveSound "grunt/active"
	Obituary "%o was chopped down by a Shotgunner."
	States
	{
	Spawn:
		ZSP1 A 10
		ZSPI A 10
		Loop
	See:
		ZSP1 AABB 3
		TNT1 A 0 A_SpawnItem("FootStep",0,0,0,0)
		ZSP1 CCDD 3
		TNT1 A 0 A_SpawnItem("FootStep",0,0,0,0)
		Loop
	Missile:
		TNT1 AAAA 0 Bright
		ZSP1 F 5 Bright
		ZSP1 E 10
		ZSP1 E 8 
		TNT1 A 0 A_SpawnItemEx("ShellCasing", -8, 0, 30, Random(2, 6), 0, Random(1, 3), Random(-80, -90), 128)
		ZSP1 E 8
		ZSP1 E 10
		Goto See
    }
}
