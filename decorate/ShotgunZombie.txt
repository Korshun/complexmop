ACTOR PossessedShotgunZombie : PossessedMonster
{ 
	Player.MorphWeapon "ShotgunZombieWeapon"

	Health 30
	Radius 20
	Height 56
	Speed 0.35
	Mass 100
	PainChance 160
	DropItem "HealthFlask" 35
	DropItem "ArmorCharge" 35
	DropItem "AmmoPack" 20
	DropItem "HandGrenadeAmmo" 50 1
	DropItem "MineAmmo" 25 1
	DropItem "Combat Shotgun" 256 1
	+FLOORCLIP
	+MISSILEMORE
	SeeSound "grunt/sight"
	AttackSound "weapons/12gaugefire"
	PainSound "grunt/pain"
	DeathSound "grunt/death"
	ActiveSound "grunt/active"
	Obituary "%o was chopped down by a Shotgunner."
	States
	{
	Spawn:
		ZSP1 A 10
		ZSPI A 10
		Loop
	See:
		ZSP1 AABB 3
		TNT1 A 0 A_SpawnItem("FootStep",0,0,0,0)
		ZSP1 CCDD 3
		TNT1 A 0 A_SpawnItem("FootStep",0,0,0,0)
		Loop
	Missile:
		TNT1 AAAA 0 Bright
		ZSP1 F 5 Bright
		ZSP1 E 10
		ZSP1 E 8
		ZSP1 E 8
		ZSP1 E 10
		Goto See
    }
}

Actor ShotgunZombieWeapon : PossessedWeapon
{
	AttackSound "weapons/12gaugefire"

	States
	{
	Ready:
		SHTG A 1 A_WeaponReady
		Loop
	Fire:
		TNT1 A 0 Radius_Quake(3,3,0,1,0)
		TNT1 A 0 A_GunFlash
		TNT1 A 0 A_Light1
		TNT1 AAAA 0 A_FireCustomMissile("MonsterTracer2",Random(6,-6),0,0,Random(6,-6))
		SHTF A 3 Bright A_FireBullets(22.5,0,5,4)
		SHTF B 2 Bright A_Light2
		TNT1 A 0 A_Light0
		SHTF CDEF 1
		SHTG BCDEFG 1
		SHTG G 2 A_PlaySound("weapons/12gsgcock")
		SHTG HI 2
		SHTG J 1 A_SpawnItemEx("ShellCasing",30,-2,25,Random(2,4),Random(3,6),Random(3,6),0)
		SHTG IHGF 1
		SHTG EDCB 1
		Goto Ready
    }
}
