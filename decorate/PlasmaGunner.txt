Actor PossessedPlasmaGunner : PossessedMonster
{
	Player.MorphWeapon "Zombie Plasma Gun"
	
	Health 20
	Radius 20
	Height 56
	Speed 0.35 // FIXME: how to convert speeds?
	Mass 100
	PainChance 160
	+FLOORCLIP
	+MISSILEMORE
	SeeSound "fem/sight"
	PainSound "fem/pain"
	DeathSound "fem/death"
	ActiveSound "fem/active"
	Obituary "%o was humiliated by a Plasma zombiewoman."
	States
	{
	Spawn:
        ZMAY A 10
		ZMAI A 10
        Loop
	See:
        ZMAY AABB 3
        TNT1 A 0 A_SpawnItem("FootStep",0,0,0,0)
        ZMAY CCDD 3 
        TNT1 A 0 A_SpawnItem("FootStep",0,0,0,0)
        Loop
	Missile:
        ZMAY F 2 Bright
        ZMAY E 3
		ZMAY F 2 Bright
        ZMAY E 3
		ZMAY F 2 Bright
        ZMAY E 3
		ZMAY E 20
		ZMAY E 10
        Goto See
    }
}

ACTOR "Zombie Plasma Gun" : PossessedWeapon
{
	States
	{
	Ready:
		PKPL A 1 A_WeaponReady
		Loop
	Fire:
		TNT1 A 0 A_GunFlash
		PLSF B 2 A_FireCustomMissile("ZombiePlasmaBall")
		PKPL A 3 Bright
		PLSF B 2 A_FireCustomMissile("ZombiePlasmaBall")
		PKPL A 3 Bright
		PLSF B 2 A_FireCustomMissile("ZombiePlasmaBall")
		PKPL A 3 Bright
		PKPL DEF 1
		PKPL G 7
		PKPL FED 2	
		PKPL CBA 1
		Goto Ready
	Flash:
		TNT1 A 2 Bright A_Light1
		PLSF A 2 Bright A_Light2
		TNT1 A 0 A_Light0
		Stop
    }
}
