ACTOR PossessedDevilImp : PossessedMonster
{
	Player.MorphWeapon "DevilImpWeapon"

	Health 60
	Radius 20
	Height 56
	Speed 0.35
	MeleeDamage 3
	PainChance 140
	DamageFactor "ImpComet", 0.0
	SeeSound "imp/sight"
	PainSound "imps/pain"
	DeathSound "imp/death"
	ActiveSound "imp/active"
	MeleeSound "imps/melee"
	Obituary "%o got burned by a Devil Imp."
	HitObituary "%o was shredded by a Devil Imp."
	+FLOORCLIP
	+MISSILEMORE
	Species "Imps"
	States
	{
	Spawn:
		SIMP A 10
		SIMI A 10
		Loop
	See:
		SIMP AABB 3
		TNT1 A 0 A_SpawnItem("HumanishStep",0,0,0,0)
		SIMP CCDD 3 
		TNT1 A 0 A_SpawnItem("HumanishStep",0,0,0,0)
		Loop
	Melee: 
		SIMP EF 5
		SIMP G 4
		SIMP FE 5 
		Goto See 
	Missile:
		SIMP EF 8
		SIMP G 4 Bright
		SIMP FEF 6
		SIMP G 4 Bright
		SIMP FE 4
		Goto See
		}
}

Actor DevilImpWeapon : PossessedWeapon
{
    States
    {
	 Ready:
		DCLW A 1 A_WeaponReady
		Loop
     Fire:
		DCLW A 8 
		SCLW B 4 Bright A_FireCustomMissile("NewImpBall", 0, 0, 0)
		DCLW AAA 6
		SCLW B 4 Bright A_FireCustomMissile("NewImpBall", 0, 0, 0)	
		DCLW AA 4
		Goto Ready
     AltFire:
		DCLW AB 5
		DCLW C 4 A_CustomPunch(3)
		DCLW BA 5 
        Goto Ready
     }
}
